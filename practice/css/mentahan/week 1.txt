4. HTML Attribute
  <p class="merah">text</p>
  attribute: class
  value dari attribute class: merah

  <meta name="keywords" content="berita, hoaks"/>
  <meta name="description" content="Kuningan dalam "bahaya"/>
  attribute: name, content
  value dari attribute name: keywords
  value dari attribute content: berita, hoaks

  <img src="bunga.jpg" alt="Bunga"/>
  attribute: src, alt
  value dari attribute src: bunga.jpg
  value dari attribute alt: Bunga
